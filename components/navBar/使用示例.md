# navBar 组件使用示例

## 在首页使用（带搜索、地址、天气）

### 1. 配置 home.json
```json
{
  "navigationStyle": "custom",
  "usingComponents": {
    "nav-bar": "/components/navBar/navBar"
  }
}
```

### 2. 在 home.wxml 中使用
```xml
<nav-bar 
  searchBar="{{true}}"
  showaw="{{true}}"
  showlogo="{{true}}"
  address="{{currentAddress}}"
  weather="{{weatherInfo}}"
  logoImage="/images/logo.png"
  background="rgba(255, 255, 255, 0.95)"
  bind:search="onSearch"
  bind:weather="onWeather"
  bind:logo="onLogo"
/>

<view class="container">
  <!-- 页面内容 -->
</view>
```

### 3. 在 home.js 中添加数据和方法
```javascript
Page({
  data: {
    currentAddress: '北京市',
    weatherInfo: '22° 晴天'
  },

  onSearch() {
    // 跳转到搜索页面
    wx.navigateTo({
      url: '/pages/search/search'
    })
  },

  onWeather() {
    // 处理天气点击
    wx.showToast({
      title: '查看天气详情',
      icon: 'none'
    })
  },

  onLogo() {
    // 处理Logo点击
    wx.switchTab({
      url: '/pages/home/home'
    })
  }
})
```

## 在详情页使用（返回+首页）

### 1. 配置页面 json
```json
{
  "navigationStyle": "custom",
  "usingComponents": {
    "nav-bar": "/components/navBar/navBar"
  }
}
```

### 2. 在 wxml 中使用
```xml
<nav-bar 
  title="详情页"
  back="{{true}}"
  home="{{true}}"
  bind:back="onBack"
  bind:home="onHome"
/>

<view class="container">
  <!-- 页面内容 -->
</view>
```

### 3. 在 js 中处理事件
```javascript
Page({
  onBack(e) {
    const delta = e.detail.delta || 1
    wx.navigateBack({ delta })
  },

  onHome() {
    wx.switchTab({
      url: '/pages/home/home'
    })
  }
})
```

## 自定义样式导航栏

```xml
<nav-bar 
  background="#667eea"
  color="#ffffff"
  iconTheme="white"
  title="自定义导航栏"
/>
```

## 完全自定义内容

```xml
<nav-bar background="#f5f5f5">
  <view slot="left">
    <text>菜单</text>
  </view>
  <view slot="center">
    <text>自定义标题</text>
  </view>
  <view slot="right">
    <text>更多</text>
  </view>
</nav-bar>
```
